实现Promise.all，进阶限制并发数

  • 实现Promise.all
function myAll(arr){
    return new Promise((res)=>{
        let result=[]
        for(let i=0;i<arr.length;i++){
            let t=async()=>{
                let r=await arr[i];
                result[i]=r;
                if(result.length===arr.length){
                    res(result);
                }
            }
            t();
        }
    })
}

  • 实现一个带并发数的Promise.all
function myAllLimit(arr,limit){
    return new Promise((res)=>{
        let result=[];
        let len=arr.length;
        for(let i=0;i<limit;i++){
            let t=async()=>{
                while(arr.length>0){
                    let j=arr.length-1;
                    let r=await arr.pop();
                    result[j]=r;
                }
                if(result.length===len){
                    res(result);
                }
            }
            t();
        }
    })
}
